总线请求形式分XML请求和JSON请求

举例：

1. soap的服务方

#####真实请求报文

POST / HTTP/1.1
Content-Type: application/xml

<soapenv:Envelope xmlns:soapenv=http://schemas.xmlsoap.org/soap/envelope/
 xmlns:ser="http://impl.services.cxf.rpc.com">
<soapenv:Header>
</soapenv:Header>
<soapenv:Body>
<ser:query>
    <condition>SG</condition>
    <sourceItems>
        <item>XM</item>
        <item>XB</item>
    </sourceItems>
    <sourceDataSet>
        <item>
            <item>XM</item>
            <item>SFZH</item>
        </item>
        <item>
            <item>李四</item>
            <item>xxxxxxxxxxxxxx</item>
        </item>
        <item>
            <item>张三</item>
            <item>YYYYYYYYYYYYYY</item>
        </item>
    <sourceDataSet>
    <infoCodeMode>0</infoCodeMode>
   </ser:query>
</soapenv:Body>
</soapenv:Envelope>

#####总线XML请求

POST / HTTP/1.1
Content-Type: application/xml

<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" >
    <soapenv:Header>

        <CyberpoliceSBReqHeader>
            <rid> S50000000000000001</rid>

    </soapenv:Header>
    <soapenv:Body>
        <soapenv:query>
            <condition>SG</condition>

                <item>XM</item>

            </sourceItems>
            <sourceDataSet>
                <item>
                    <item>XM</item>

                    <item>李四</item>

                </item>
            </sourceDataSet>
            <infoCodeMode>0</infoCodeMode>
        </soapenv:query>
    </soapenv:Body>
</soapenv:Envelope>

Content-Type: application/json

#####总线请求JOSN形式

POST / HTTP/1.1   
Content-Type: application/json
requestmethod: GET 
pathparams: /service/query?paramOne=1&paramTwo=2

ridpasswd: 0

retrynum: 1

}
